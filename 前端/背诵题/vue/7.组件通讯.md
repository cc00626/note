### 父子组件通讯

**props和emit**

```
//父组件
<template>
  <div>
    <h2>父组件</h2>
    <p>当前消息：{{ parentMsg }}</p>
    <Child :child-msg="parentMsg" @update-msg="handleUpdate" />
  </div>
</template>

<script setup>
import { ref } from 'vue';
import Child from './Child.vue';

const parentMsg = ref('Hello 从父组件来');

const handleUpdate = (newVal) => {
  parentMsg.value = newVal; // 接收子组件传来的值并更新
};
</script>

//子组件
<template>
  <div style="border: 1px solid red; padding: 10px;">
    <h3>子组件</h3>
    <p>收到父组件的话：{{ childMsg }}</p>
    <button @click="sendBack">回复父组件</button>
  </div>
</template>

<script setup>
// 1. 接收 Props
defineProps({
  childMsg: String
});

// 2. 声明 Emits
const emit = defineEmits(['update-msg']);

const sendBack = () => {
  // 触发事件并传递数据
  emit('update-msg', '这是子组件的回复！');
};
</script>
```



**ref和defineExpose**

```
//子组件
<template>
  <div v-if="isVisible" class="modal">
    <h3>我是子组件弹窗</h3>
    <p>当前状态：{{ status }}</p>
    <button @click="isVisible = false">关闭</button>
  </div>
</template>

<script setup>
import { ref } from 'vue';

const isVisible = ref(false);
const status = ref('待命');

// 定义一个方法，供父组件调用
const open = (msg) => {
  isVisible.value = true;
  status.value = msg || '被父组件打开了';
  console.log('子组件：收到指令，已打开！');
};

// 【关键步骤】显式暴露属性和方法
defineExpose({
  open,
  status
});
</script>

<style scoped>
.modal {
  border: 2px solid #42b983;
  padding: 20px;
  margin-top: 10px;
}
</style>


//父组件
<template>
  <div class="parent">
    <h2>父组件控制台</h2>
    
    <Child ref="childRef" />

    <hr />
    
    <button @click="callChild">通过 Ref 打开子组件</button>
    <button @click="checkChildStatus">查看子组件状态</button>
  </div>
</template>

<script setup>
import { ref } from 'vue';
import Child from './Child.vue';

// 2. 创建一个同名的 ref 对象，初始为 null
// 注意：变量名必须与模板中的 ref="childRef" 完全一致
const childRef = ref(null);

const callChild = () => {
  // 3. 通过 .value 访问子组件暴露的方法
  childRef.value.open('我是父组件传来的悄悄话');
};

const checkChildStatus = () => {
  // 也可以访问子组件暴露的响应式变量
  alert('子组件当前状态是：' + childRef.value.status);
};
</script>
```



### 祖孙通讯

project，inject

穿透属性

```
//父组件
<template>
  <MyInput 
    label="用户名" 
    placeholder="请输入" 
    type="password"
    class="custom-border"
    @focus="handleFocus"
  />
</template>

<script setup>
import MyInput from './MyInput.vue';
const handleFocus = () => console.log('输入框聚焦了！');
</script>


//子组件
<template>
  <div class="my-input-wrapper">
    <label>{{ label }}</label>
    
    <input v-bind="$attrs" class="real-input" />
  </div>
</template>

<script setup>
// 显式声明 label，剩下的 type, placeholder, @focus 都会进入 $attrs
defineProps(['label']);

// 如果你不希望组件根元素 (div) 自动继承这些属性，可以开启这个配置：
// defineOptions({ inheritAttrs: false });
</script>

<style scoped>
	.my-input-wrapper { display: flex; flex-direction: column; }
	.real-input { padding: 8px; border: 1px solid #ccc; }
	.custom-border { border-color: blue !important; } /* class 也会透传进来 */
</style>
```
