### 自定义指令

```
const vPermission = {
  mounted(el, binding) {
    const { value } = binding; // 获取指令绑定的权限值
    const userRoles = ['admin', 'editor']; // 模拟当前用户的角色

    if (value && !userRoles.includes(value)) {
      // 如果用户没有权限，直接从父节点移除该 DOM
      el.parentNode && el.parentNode.removeChild(el);
    }
  }
}

// 模板使用
// <button v-permission="'admin'">只有管理员可见</button>
```
