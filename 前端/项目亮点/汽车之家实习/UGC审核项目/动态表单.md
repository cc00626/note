```
import React from 'react';
import { Form, Input, Select, DatePicker, Button, InputNumber, Checkbox } from 'antd';
import { useEffect } from 'react';
// 控件映射表：根据 type 匹配对应的 antd 组件
const ComponentMap = {
  input: Input,
  password: Input.Password,
  select: Select,
  number: InputNumber,
  date: DatePicker,
  checkbox: Checkbox,
  textarea: Input.TextArea,
};

const DynamicForm = ({ schema, onFinish, initialValues,data }) => {
  const [form] = Form.useForm();
// 监听 data 的变化，一旦数据从后端回来，立刻填充表单
  useEffect(() => {
    if (data) {
      form.setFieldsValue(data); // 关键点：手动更新表单状态
    }
  }, [data, form]);
  return (
    <Form
      form={form}
      layout="vertical"
      initialValues={initialValues}
      onFinish={onFinish}
      style={{ maxWidth: 600 }}
    >
      {schema.fields.map((item) => {
        const { type, name, label, rules, props, options } = item;
        const Component = ComponentMap[type] || Input; // 找不到类型默认用 Input

        return (
          <Form.Item key={name} name={name} label={label} rules={rules}>
            {/* 渲染对应的组件，并透传 props */}
            <Component {...props} options={options} placeholder={props?.placeholder || `请输入${label}`} />
          </Form.Item>
        );
      })}

      <Form.Item>
        <Button type="primary" htmlType="submit">提交审核</Button>
        <Button style={{ marginLeft: 8 }} onClick={() => form.resetFields()}>重置</Button>
      </Form.Item>
    </Form>
  );
};

export default DynamicForm;

```



## 字段之间的关联联动怎么管理 ?

在审核平台中，在一些内容审核不通过，需要在下方展示不通过的原因，字段之间需要进行关联和联动。

通过一个表单项目dependencies字段进行关联。

在渲染的时候会遍历获取依赖的关系，将他们放入一个Map数据中。左边表示字段，右边表示受影响的字段。

```
A → [B, C]
B → [D]
C → []
D → []
```

当某个字段的值发生变化的时候，通过这个Map数据

### 找到所有依赖这个字段的节点(直接和间接的)

通过广度优先遍历，遍历字段。如果字段在Map结构数据中，找到，就添加到结果。

```
  getAffectedFields(field) {
    const affected = new Set();
    const queue = [field];

    while (queue.length > 0) {
      const current = queue.shift();
      const neighbors = this.adjacencyList.get(current);

      if (neighbors) {
        neighbors.forEach((neighbor) => {
          if (!affected.has(neighbor)) {
            affected.add(neighbor);
            queue.push(neighbor);
          }
        });
      }
    }

    return Array.from(affected);
  }
```

### 用拓扑排序得到更新顺序

用找到受影响的字段和Map数据，更新得到更新顺序。

代码首先创建了两个 Map：`inDegree`（入度表）和 `subGraph`（局部邻接表）。

遍历受影响的字段，去Map结构中找到有没有指向关系。如果在受影响的字段里面，这个受影响的字段度 + 1。

先找到度为 0 的放入队列中为第一批要更新的结果，然后处理局部图。知道找到所有的结果。

```javascript
  getSortedFields(fields) {
    const inDegree = new Map();
    const subGraph = new Map();

    fields.forEach((field) => {
      inDegree.set(field, 0);
      subGraph.set(field, new Set());
    });

    fields.forEach((field) => {
      const neighbors = this.adjacencyList.get(field);

      if (neighbors) {
        neighbors.forEach((neighbor) => {
          if (fields.includes(neighbor)) {
            subGraph.get(field).add(neighbor);
            inDegree.set(neighbor, (inDegree.get(neighbor) || 0) + 1);
          }
        });
      }
    });

    const queue = [];
    const result = [];

    inDegree.forEach((degree, field) => {
      if (degree === 0) queue.push(field);
    });

    while (queue.length > 0) {
      const current = queue.shift();
      result.push(current);

      const neighbors = subGraph.get(current);

      if (neighbors) {
        neighbors.forEach((neighbor) => {
          const newDegree = (inDegree.get(neighbor) || 0) - 1;

          inDegree.set(neighbor, newDegree);

          if (newDegree === 0) {
            queue.push(neighbor);
          }
        });
      }
    }

    return result;
  }
```

## 表单回显怎么回显

正常的普通字段，通过在useEffect里获取初始数据接口进行回显。

有些字段没有值，需要显示默认值。通过在defautlValue里面设置默认值，回显的时候会先用默认值，没有默认值就用后端返回的值。



需要注意有些数据类型的数据需要设置数据回显。比如日期类型，需要包装成dayjs对象。

## 动态表单如何实现的？

在我们的UGC审核平台中，不同类型的帖子需要展示不同的审核表单，并且随着ai的不断训练，设计到审核规则的变化，需要频繁修改一些字段的内容。

在调研的过程中，我调研了我调研了几种方案，例如 Formily、JSON Schema Form 等动态表单方案。希望引入过重的依赖库，我们最终决定 自己实现一套轻量级的动态表单系统。

所以我们做了一套动态表单系统。通过JSON Schema配置表单结构，前端负责解析配置渲染组件，这样只需要修改配置就能够完成表单的修改。

在页面初始化的时候，后端会返回schema，和表单的初始数据。在前端通过遍历schema，根据type渲染对应的组件，然后加载初始化数据渲染表单。在用户编辑的时候通过form.setFieldsValue()设置数据，通过字段依赖图控制其他表单的显示隐藏。最后将数据提交给后端。

## 为什么不适用formily，而是自己写？

1. formily的功能更加完善，但是会增加打包后的体积，影响页面的性能。
2. 我们的系统表单的联动和校验规则并不是很复杂，主要是为了解决频繁增改字段的问题。
3. 我们人工审核页面的字段变化比较频繁，比如新增审核标签、调整审核项，如果使用 Formily 的 schema DSL（例如 x-component、x-reactions），学习 成本高，我们自己实现了一套更简单的，方便后端或者其他人员修改表单。

## 如何实现字段之间的关联？

 在我们的 **UGC 审核平台**中，不同类型的内容在审核不通过时，需要在下方展示对应的 **不通过原因**。如果每次字段变化都 **重新计算整个表单所有字段状态**，在字段很多时性能会比较差，而且逻辑也会变得复杂。所以我们设计了一套 **字段依赖图（FieldGraph）机制** 来管理字段之间的依赖关系。



在schema中，我们通过dependencies字段进行关联字段的依赖关系，在初始化遍历表单的时候，会遍历dependencies得到一个字段依赖关系图，通过一个Map结构来存储。



我们是通过拓扑排序来实现表单之间的关联的，我用了连个数据结构，一个用来保存每个字段的入度，一个用来保留每个字段关联的子图。

后见依赖关系，从初始化的map节点中，获取每个字段依赖的节点，然后判断这些节点在不在要更新的项目里，我们只需要保存需要的，然后每发现一个，被关联的节点的度 + 1，



然后遍历，将度为0 的节点找出来，他的子图所对应的度 减1，然后找出度为0的，最后依此循环。找出更新的顺序。



## 如何实现表单的校验

在schema上一个rules字段进行校验，在渲染的时候我们会将规则，对于常见的规则，我们维护一个映射函数，将常见的规则放入。

如果是属于复杂的函数僬侥场景，使用直接传入Validator 函数，进行校验。
